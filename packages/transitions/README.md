## Installation

```bash
yarn add @junipero/transitions
```

## Usage

```javascript
import { slideInUpMenu } from '@junipero/transitions';
import { Tooltip } from '@junipero/react';

export default () => (
  <Tooltip animate={slideInUpMenu}>
    Info
  </Tooltip>
);
```
